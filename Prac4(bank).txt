class Bank():
    def __init__(self):
        self.balance = 0
        print("Devansh\n")
        print("Your accout has been created!")
    
    def deposit(self):
        amount = float(input("Enter amount to be deposited: "))
        self.balance = self.balance + amount
        print("Your new balance is: %f"%self.balance)

    def withdraw(self):
        amount = float(input("Enter amount to be withdrawn: "))
        if amount <= self.balance:
            self.balance = self.balance - amount
            print(amount, "has been deducted from your account and your new balance is: %f"%self.balance)
        else:
            print("Insufficient balance!")
        
    def enquiry(self):
        print("Your new balance is: %f"%self.balance)

acc = Bank()
acc.deposit()
acc.withdraw()
acc.enquiry()
